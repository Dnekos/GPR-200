void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    
    // texture space = (0,1]
    // if texture = 100x100, resInv = 0.01,0.01
    vec2 inverseReso = 1.0 / iChannelResolution[0].xy;
    
    //for gaussian
    
    float invKernalSum = 0.125; // 1/64, 64 being the sum of the full 4x4 matrix
    vec4 kernel = vec4(invKernalSum, invKernalSum * 3.0, invKernalSum * 3.0, invKernalSum);

	vec4[4] pixelsx = vec4[4]( getPixel(fragCoord.x - 1.0,fragCoord.y, inverseReso, iChannel0),
                getPixel(fragCoord.x,fragCoord.y, inverseReso, iChannel0),
                getPixel(fragCoord.x + 1.0,fragCoord.y, inverseReso, iChannel0),
                getPixel(fragCoord.x + 2.0,fragCoord.y, inverseReso, iChannel0)
                );

	color4 blur = pixelsx[0] * kernel[0]+
        pixelsx[1] * kernel[1]+
        pixelsx[2] * kernel[2]+
        pixelsx[3] * kernel[3];
    
    
    //Edge Detection
    
    /*
	vec3 kernel = vec3(1.0,-2.0,1.0);
	
	vec4[3] pixelsx = vec4[3]( getPixel(fragCoord.x - 0.5,fragCoord.y, inverseReso, iChannel0),
                getPixel(fragCoord.x + 0.5,fragCoord.y, inverseReso, iChannel0),
                getPixel(fragCoord.x + 1.5,fragCoord.y, inverseReso, iChannel0)
                );

    color4 blur = pixelsx[0] * kernel[0]+
        pixelsx[1] * kernel[1]+
        pixelsx[2] * kernel[2];
	*/

    
    
    fragColor = blur;
}
